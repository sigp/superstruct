6/// Convert an identifier from CamelCase to snake_case.
7pub fn snake_case(ident: &str) -> String {
8    ident
9        .chars()
10        .enumerate()
11        .flat_map(|(i, c)| {
12            let chars: SmallVec<[char; 2]> = if c.is_uppercase() {
13                if i == 0 {
14                    c.to_lowercase().collect()
15                } else {
16                    std::iter::once('_').chain(c.to_lowercase()).collect()
17                }
18            } else {
19                smallvec![c]
20            };
21            chars
22        })
23        .collect()
24}

26/// Create a generics block like `<_, _, _>` with `num_generics` underscores.
27pub fn underscore_generics(num_generics: usize) -> proc_macro2::TokenStream {
28    let underscore = Token![_](Span::call_site());
29    let underscores = std::iter::repeat_n(quote! { #underscore }, num_generics);
30    quote! { <#(#underscores),*> }
31}
