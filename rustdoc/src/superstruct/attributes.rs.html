1//! Utilities to help with parsing configuration attributes.
2use darling::{export::NestedMeta, Error, FromMeta};
3use syn::Ident;
4
5/// Parse a list of nested meta items.
6///
7/// Useful for passing through attributes intended for other macros.
8#[derive(Debug)]
9pub struct NestedMetaList {
10    pub metas: Vec<NestedMeta>,
11}
12
13impl FromMeta for NestedMetaList {
14    fn from_list(items: &[NestedMeta]) -> Result<Self, Error> {
15        Ok(Self {
16            metas: items.to_vec(),
17        })
18    }
19}

21/// List of identifiers implementing `FromMeta`.
22///
23/// Useful for imposing ordering, unlike the `HashMap` options provided by `darling`.
24#[derive(Debug, Default, Clone)]
25pub struct IdentList {
26    pub idents: Vec<Ident>,
27}
28
29impl FromMeta for IdentList {
30    fn from_list(items: &[NestedMeta]) -> Result<Self, Error> {
31        let idents = items
32            .iter()
33            .map(|nested_meta| {
34                let meta = match nested_meta {
35                    NestedMeta::Meta(m) => m,
36                    NestedMeta::Lit(l) => {
37                        return Err(Error::custom(format!("expected ident, got literal: {l:?}")))
38                    }
39                };
40                let path = meta.path();
41                path.get_ident()
42                    .cloned()
43                    .ok_or(Error::custom(format!("can't parse as ident: {path:?}")))
44            })
45            .collect::<Result<_, _>>()?;
46        Ok(Self { idents })
47    }
48}
